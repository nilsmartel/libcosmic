cosmic::widget

Module container

source
Expand description

Containers let you align a widget inside their boundaries.

§Example

use iced::widget::container;

enum Message {
    // ...
}

fn view(state: &State) -> Element<'_, Message> {
    container("This text is centered inside a rounded box!")
        .padding(10)
        .center(800)
        .style(container::rounded_box)
        .into()
}
